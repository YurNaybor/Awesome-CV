%-------------------------------------------------------------------------------
% CONFIGURATIONS
%-------------------------------------------------------------------------------
% A4 paper size by default, use 'letterpaper' for US letter
\documentclass[11pt, a4paper]{awesome-cv}

% Configure page margins with geometry
\geometry{left=1.4cm, top=.8cm, right=1.4cm, bottom=1.8cm, footskip=.5cm}

% Specify the location of the included fonts
\fontdir[fonts/]

% Color for highlights
% Awesome Colors: awesome-emerald, awesome-skyblue, awesome-red, awesome-pink, awesome-orange
%                 awesome-nephritis, awesome-concrete, awesome-darknight
$if(color)$
  \colorlet{awesome}{$color$}
$else$
  \colorlet{awesome}{awesome-red}
$endif$
% Uncomment if you would like to specify your own color
% \definecolor{awesome}{HTML}{CA63A8}

% Colors for text
% Uncomment if you would like to specify your own color
% \definecolor{darktext}{HTML}{414141}
% \definecolor{text}{HTML}{333333}
% \definecolor{graytext}{HTML}{5D5D5D}
% \definecolor{lighttext}{HTML}{999999}

% Set false if you don't want to highlight section with awesome color
\setbool{acvSectionColorHighlight}{true}

% If you would like to change the social information separator from a pipe (|) to something else
\renewcommand{\acvHeaderSocialSep}{\quad\textbar\quad}


%-------------------------------------------------------------------------------
%	PERSONAL INFORMATION
%	Comment any of the lines below if they are not required
%-------------------------------------------------------------------------------
% Available options: circle|rectangle,edge/noedge,left/right
% \photo[circle,noedge,left]{./examples/profile}
$if(photo)$\photo[$photo.mode$]{$photo.location$}$endif$
\name{$name.first$}{$name.last$}
$if(position)$
  \position{
    $for(position)$
      $position$$sep${\enskip\cdotp\enskip}
    $endfor$
  }
$endif$
$if(address)$
  \address{
    $for(address)$
      $address$$sep$,
    $endfor$
  }
$endif$

$if(phone)$\mobile{$phone$}$endif$
$if(email)$\email{$email$}$endif$
$if(homepage)$\homepage{$homepage$}$endif$
$if(github)$\github{$github$}$endif$
$if(gitlab)$\gitlab{$gitlab$}$endif$
$if(stackoverflow)$\stackoverflow{$stackoverflow.id$}{$stackoverflow.name$}$endif$
$if(linkedin)$\linkedin{$linkedin$}$endif$
$if(twitter)$\twitter{$twitter$}$endif$
$if(skype)$\skype{$skype$}$endif$
$if(reddit)$\reddit{$reddit$}$endif$
$if(extrainfo)$\extrainfo{$extrainfo$}$endif$

$if(quote)$\quote{``$quote$"}$endif$

%-------------------------------------------------------------------------------
%	LETTER INFORMATION
%	All of the below lines must be filled out
%-------------------------------------------------------------------------------
% The company being applied to
\recipient
  {$recipient.name$}
  {$for(recipient.address)$$recipient.address$$sep$\\$endfor$}
% The date on the letter, default is the date of compilation
\letterdate{\today}
% The title of the letter
\lettertitle{$letter.job$}
% How the letter is opened
\letteropening{$letter.opening$}
% How the letter is closed
\letterclosing{$if(letter.closing)$$letter.closing$$else$Sincerely,$endif$}
% Any enclosures with the letter
$if(letter.attachments)$
  \letterenclosure[Attached]{$for(letter.attachments)$$letter.attachments$$sep$, $endfor$}
$endif$


%-------------------------------------------------------------------------------
\begin{document}

% Print the header with above personal informations
% Give optional argument to change alignment(C: center, L: left, R: right)
\makecvheader[R]

% Print the footer with 3 arguments(<left>, <center>, <right>)
% Leave any of these blank if they are not needed
\makecvfooter
  {\today}
  {$name.first$ $name.last$~~~Â·~~~Cover Letter}
  {}

% Print the title with above letter informations
\makelettertitle

%-------------------------------------------------------------------------------
%	LETTER CONTENT
%-------------------------------------------------------------------------------
\begin{cvletter}
  $for(sections)$
    $if(pretty)$\lettersection{$sections.name$}$endif$
    $sections.text$
  $endfor$
\end{cvletter}


%-------------------------------------------------------------------------------
% Print the signature and enclosures with above letter informations
\makeletterclosing

\end{document}
