%-------------------------------------------------------------------------------
\begin{document}

% Print the header with above personal informations
% Give optional argument to change alignment(C: center, L: left, R: right)

% Leave any of these blank if they are not needed
\makecvfooter
  {$if(date)$\today$endif$}
  {$name.first$ $name.last$~~~·~~~Curriculum Vitae}
  {\thepage}


%-------------------------------------------------------------------------------
%	CV/RESUME CONTENT
%-------------------------------------------------------------------------------

%% Summary
$if(summary)$
  \cvsection{Zusammenfassung}
  \begin{cvparagraph}
    % --- Important empty line

    $summary$
  \end{cvparagraph}
$endif$

%% Experience
$if(experience)$
  \cvsection{Berufserfahrung}
  \begin{cventries}
    % --- Important empty line

    $for(experience)$
      \cventry
      {$experience.position$}
      {$experience.employer$}
      {$experience.location$}
      {$experience.period$}
      {
        $if(experience.roles)$
          \begin{cvitems}
            $for(experience.roles)$
              \item {$experience.roles$}
            $endfor$
          \end{cvitems}
        $endif$
      }
      % --- Important empty line

    $endfor$
  \end{cventries}
$endif$

%% Education
$if(education)$
  \cvsection{Ausbildung}
  \begin{cventries}
    $for(education)$
      \cventry
      {$education.subject$}
      {$education.institute$}
      {$education.location$}
      {$education.period$}
      {
        $if(education.modules)$
          \begin{cvitems}
            $for(education.modules)$
              \item {$education.modules$}
            $endfor$
          \end{cvitems}
        $endif$

    $endfor$
  \end{cventries}
$endif$

%% Qualification
$if(qualification)$
  \cvsection{Weiterqualifizierung}
  \begin{cventries}
    $for(qualification)$
      \cventry
      {$qualification.subject$}
      {$qualification.institute$}
      {$qualification.location$}
      {$qualification.period$}
      {
        $if(qualification.modules)$
          \begin{cvitems}

        $endif$

    $endfor$
  \end{cventries}
$endif$

%% Skills
$if(skills)$
  \cvsection{Fähigkeiten}
  \begin{cvskills}
    $for(skills)$
      \cvskill
      {$skills.topic$}
      {
        $for(skills.list)$
          $skills.list$$sep$,
        $endfor$
      }
    $endfor$
  \end{cvskills}
$endif$

%% Languages
$if(languages)$
  \cvsection{Sprachen}
  \begin{cvskills}
    $for(languages)$
      \cvskill
      {$languages.language$}
      {$languages.proficiency$}
    $endfor$
  \end{cvskills}
$endif$

%% Publications
$if(publications)$
  \cvsection{Publications}
  \begin{cventries}
    $for(publications)$
      \cventry
      {$publications.author$}
      {$publications.title$}
      {$publications.year$}
      {}
      {In: \emph{$publications.booktitle$}. Vol. $publications.volume$. $publications.series$. $publications.publisher$, pp. $publications.pages$}
    $endfor$
  \end{cventries}
$endif$

%% Extracurricular
$if(extra)$
  \cvsection{Extracurricular Activity}
  \begin{cventries}
    $for(extra)$

      {$extra.role$}

      {$extra.location$}
      {$extra.period$}
      {
        $if(extra.description)$

        {$honors.awards.position$}

  $endfor$
$endif$

  \cvsection{Presentation}

        $if(presentation.description)$

    $for(committees)$

  \cvsection{Interessen}

    % --- Important empty line
